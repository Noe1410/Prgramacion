public class TokiPakala {

  public static void main(String[] args) {
    String letra = "";
    String letraV = "";
    int longitudDocumento = (int) (Math.random() * 21) + 1;

    for (int documento = 1; documento <= longitudDocumento; documento++) { //creacion de documento
      int longitudParrafo = (int) (Math.random() * 12) + 1;

      for (int parrafo = 1; parrafo <= longitudParrafo; parrafo++) { //creacion de parrafos
        int longitudFrase = (int) (Math.random() * 53) + 1;

        for (int frase = 1; frase <= longitudFrase; frase++) { //creacion de frases
          int longitudPalabra = (int) (Math.random() * 6) + 1;

          for (int palabra = 1; palabra <= longitudPalabra; palabra++) { //creacion de palabras
            int probabilidadLetra = (int) (Math.random() * 8) + 1;
            int vocales = (int) (Math.random() * 5) + 1;
            int consonantes = (int) (Math.random() * 9) + 1;

            if (probabilidadLetra == 1) { //silaba formada por una sola letra
              switch (vocales) {
                case 1:
                  letra = "a";
                  break;
                case 2:
                  letra = "e";
                  break;
                case 3:
                  letra = "i";
                  break;
                case 4:
                  letra = "o";
                  break;
                case 5:
                  letra = "u";
                  break;
              }

              System.out.print(letra);
            } else { //silaba formado por consonante y vocal
              switch (consonantes) {
                case 1:
                  letra = "p";
                  break;
                case 2:
                  letra = "t";
                  break;
                case 3:
                  letra = "k";
                  break;
                case 4:
                  letra = "s";
                  break;
                case 5:
                  letra = "m";
                  break;
                case 6:
                  letra = "n";
                  break;
                case 7:
                  letra = "l";
                  break;
                case 8:
                  letra = "j";
                  break;
                case 9:
                  letra = "w";
                  break;
              }

              switch (vocales) {
                case 1:
                  letraV = "a";
                  break;
                case 2:
                  letraV = "e";
                  break;
                case 3:
                  letraV = "i";
                  break;
                case 4:
                  letraV = "o";
                  break;
                case 5:
                  letraV = "u";
                  break;
              }

              System.out.print(letra + letraV);
            }
          } //final creacion de palabras

          if (frase < longitudFrase) {
            System.out.print(" "); //espacio entre palabras
          }
        } //final creacion de frases

        System.out.print("."); //final frase
        System.out.print(" "); //incio de una nueva frase
      } //final parrafo

      System.out.printf("\n\n\n"); //retorno de linea doble
    } //final creacion documento
  }
}
